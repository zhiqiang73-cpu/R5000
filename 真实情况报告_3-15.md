```python
def calculate_position_with_dynamic_leverage(
    grade, stop_distance, account_balance, current_price, confidence
):
    # 根据信心度动态调整杠杆
    if confidence > Decimal("0.8"):
        leverage = 3
        risk_pct = Decimal("0.02")  # 2%
    elif confidence > Decimal("0.6"):
        leverage = 2
        risk_pct = Decimal("0.015")  # 1.5%
    else:
        leverage = 1
        risk_pct = Decimal("0.01")  # 1%

    risk_usd = account_balance * risk_pct
    position = risk_usd / stop_distance
    max_position = account_balance * leverage * Decimal("0.8") / current_price
    return min(position, max_position)
```
